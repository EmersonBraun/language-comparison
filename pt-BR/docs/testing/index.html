package math

import "testing"

func TestAdd(t *testing.T) {
  result := Add(1, 2)
  if result != 3 {
      t.Errorf("Add(1, 2) = %d; want 3", result)
  }
}

func TestNegative(t *testing.T) {
  result := Add(-1, 1)
  if result != 0 {
      t.Errorf("Add(-1, 1) = %d; want 0", result)
  }
}

// Table-driven tests (Go idiom)
func TestAddTable(t *testing.T) {
  tests := []struct {
      a, b, want int
  }{
      {1, 2, 3},
      {-1, 1, 0},
      {0, 0, 0},
  }
  for _, tt := range tests {
      got := Add(tt.a, tt.b)
      if got != tt.want {
          t.Errorf("Add(%d, %d) = %d; want %d",
              tt.a, tt.b, got, tt.want)
      }
  }
}

type UserRepo interface {

type MockUserRepo struct {

func (m *MockUserRepo) Find(id int) (*User, error) {
  return m.FindFunc(id)
}

func TestGetUser(t *testing.T) {
  mock := &MockUserRepo{
      FindFunc: func(id int) (*User, error) {
          return &User{Name: "John"}, nil
      },
  }

  if err != nil {
      t.Fatal(err)
  }
  if user.Name != "John" {
      t.Errorf("got %s; want John", user.Name)
  }
}

Key Takeaways​

  • Built-in vs external frameworks -- Rust, Go, and Zig ship with test runners; JavaScript, Python, PHP, and C++ rely on external tools (Jest, pytest, PHPUnit, Google Test).
  • Assertion styles differ -- Go uses manual if got != want { t.Errorf(...) }; xUnit/Jest use Assert.Equal/expect().toBe; RSpec and FluentAssertions favor BDD-style expect(x).to eq(y).
  • Mocking approaches vary -- Interface-based mocking is common in Go and Swift; Java/C# use Mockito/Moq; Python uses unittest.mock; Rust uses crates like mockall with trait-based mocks.
  • Parameterized tests are idiomatic in some ecosystems -- Go favors table-driven tests; xUnit uses [Theory] and [InlineData]; JUnit 5 and Swift Testing have @ParameterizedTest and @Test(arguments:).
  • Performance testing is built-in for Go -- go test -bench=. runs benchmarks; other languages typically use separate tools (pytest-benchmark, criterion for Rust, etc.).
